import React, { useState, useEffect } from 'react';

// --- Helper Data & Configuration ---

const verbs = [
  { value: 'avoir', label: 'avoir (to have)' },
  { value: 'être', label: 'être (to be)' },
  { value: 'faire', label: 'faire (to do/make)' },
  { value: 'aller', label: 'aller (to go)' },
  { value: 'dire', label: 'dire (to say/tell)' },
];

const tenses = [
  { value: 'Présent', label: 'Present (Le présent)' },
  { value: 'Passé Composé', label: 'Present Perfect (Passé Composé)' },
  { value: 'Imparfait', label: 'Past Continuous (Imparfait)' },
  { value: 'Plus-que-parfait', label: 'Past Perfect (Plus-que-parfait)' },
  { value: 'Futur Simple', label: 'Future Simple (Futur simple)' },
  { value: 'Futur Antérieur', label: 'Future Perfect (Futur antérieur)' },
  // Add other tenses as needed, ensuring the prompt can handle them
];

// --- Main App Component ---

export default function App() {
  const [selectedVerb, setSelectedVerb] = useState(verbs[0].value);
  const [selectedTense, setSelectedTense] = useState(tenses[0].value);
  const [quizData, setQuizData] = useState([]);
  const [userAnswers, setUserAnswers] = useState({});
  const [quizState, setQuizState] = useState('config'); // 'config', 'loading', 'active', 'results'
  const [currentQuestionIndex, setCurrentQuestionIndex] = useState(0);

  // --- API Call to Generate Quiz ---
  const generateQuiz = async () => {
    setQuizState('loading');
    setQuizData([]);
    setUserAnswers({});
    setCurrentQuestionIndex(0);

    const prompt = `
      Create a French language multiple-choice quiz with exactly 20 questions.
      The quiz must focus on testing the user's ability to identify the correct tense for the verb "${selectedVerb}".
      The target tense for all questions is: "${selectedTense}".

      For each of the 20 questions:
      1.  Randomly pick a subject pronoun (e.g., Je, Tu, Il, Elle, Nous, Vous, Ils, Elles).
      2.  Create an English question like "Which form means '[pronoun] [verb in English]'?". For example: "Which form means 'I have'?" or "Which form means 'they are'?".
      3.  Generate the single correct French answer for that pronoun in the target tense, "${selectedTense}".
      4.  Generate three plausible but incorrect "distractor" options. Each distractor MUST be a conjugation of the same verb ("${selectedVerb}") and the same pronoun, but in a DIFFERENT tense (e.g., Imparfait, Passé Composé, Futur Simple, etc.). This is critical.
      5.  Provide a "hint" which is simply the name of the target tense: "${selectedTense}".

      Provide the response as a JSON array of objects. Each object must have five keys: "id" (number), "question" (the English question), "options" (an array of all 4 French options, shuffled), "correctAnswer" (the single correct French option), and "hint" (the target tense name).

      Example for verb "avoir", target tense "Présent":
      [
        {
          "id": 1,
          "question": "Which form means 'I have'?",
          "options": ["J'ai eu", "J'avais", "J'aurai", "J'ai"],
          "correctAnswer": "J'ai",
          "hint": "Présent"
        },
        {
          "id": 2,
          "question": "Which form means 'we have'?",
          "options": ["Nous avions", "Nous avons", "Nous aurons", "Nous avons eu"],
          "correctAnswer": "Nous avons",
          "hint": "Présent"
        }
      ]
    `;

    try {
        let chatHistory = [{ role: "user", parts: [{ text: prompt }] }];
        const payload = { 
            contents: chatHistory,
            generationConfig: {
                responseMimeType: "application/json",
                responseSchema: {
                    type: "ARRAY",
                    items: {
                        type: "OBJECT",
                        properties: {
                            "id": { "type": "NUMBER" },
                            "question": { "type": "STRING" },
                            "options": { 
                                "type": "ARRAY",
                                "items": { "type": "STRING" }
                            },
                            "correctAnswer": { "type": "STRING" },
                            "hint": { "type": "STRING" }
                        },
                        required: ["id", "question", "options", "correctAnswer", "hint"]
                    }
                }
            }
        };
        const apiKey = ""; // API key will be handled by the environment
        const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

        const response = await fetch(apiUrl, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(payload)
        });

        if (!response.ok) {
            throw new Error(`API call failed with status: ${response.status}`);
        }
        
        const result = await response.json();

        if (result.candidates && result.candidates[0].content && result.candidates[0].content.parts[0]) {
            const rawJson = result.candidates[0].content.parts[0].text;
            const parsedQuiz = JSON.parse(rawJson);
            parsedQuiz.forEach(q => {
                q.options = q.options.sort(() => Math.random() - 0.5);
            });
            setQuizData(parsedQuiz);
            setQuizState('active');
            setUserAnswers(Object.fromEntries(parsedQuiz.map(q => [q.id, null])));
        } else {
            throw new Error("Invalid response structure from API.");
        }
    } catch (error) {
        console.error("Error generating quiz:", error);
        setQuizState('config');
    }
  };

  // --- Event Handlers ---
  const handleAnswerSelect = (questionId, answer) => {
    setUserAnswers(prev => ({ ...prev, [questionId]: answer }));
  };

  const handleNextQuestion = () => {
    if (currentQuestionIndex < quizData.length - 1) {
      setCurrentQuestionIndex(prev => prev + 1);
    } else {
      setQuizState('results');
    }
  };
  
  const handleStartOver = () => {
    setQuizState('config');
  };

  // --- Render Logic ---

  const renderContent = () => {
    switch (quizState) {
      case 'loading':
        return <LoadingSpinner />;
      case 'active':
        return (
          <QuizView
            questionData={quizData[currentQuestionIndex]}
            currentIndex={currentQuestionIndex}
            totalQuestions={quizData.length}
            userAnswer={userAnswers[quizData[currentQuestionIndex].id]}
            onAnswerSelect={handleAnswerSelect}
            onNext={handleNextQuestion}
          />
        );
      case 'results':
        return (
            <ResultsDisplay 
                quizData={quizData} 
                userAnswers={userAnswers}
                onStartOver={handleStartOver}
            />
        );
      case 'config':
      default:
        return (
          <ConfigurationScreen
            selectedVerb={selectedVerb}
            setSelectedVerb={setSelectedVerb}
            selectedTense={selectedTense}
            setSelectedTense={setSelectedTense}
            onStartQuiz={generateQuiz}
          />
        );
    }
  };

  return (
    <div className="bg-slate-900 text-slate-200 min-h-screen font-sans flex flex-col items-center justify-center p-4">
      <div className="w-full max-w-2xl mx-auto">
        {quizState !== 'config' && <Header />}
        <main className="bg-slate-800 rounded-xl shadow-2xl p-6 sm:p-8 mt-6 ring-1 ring-slate-700">
          {renderContent()}
        </main>
        {quizState === 'config' && <Footer />}
      </div>
    </div>
  );
}

// --- UI Components ---

const Header = () => (
  <header className="text-center mb-4">
    <h1 className="text-3xl sm:text-4xl font-bold text-white">French Verb Master</h1>
  </header>
);

const ConfigurationScreen = ({ selectedVerb, setSelectedVerb, selectedTense, setSelectedTense, onStartQuiz }) => (
  <div className="flex flex-col gap-6 animate-fade-in text-center">
     <Header />
     <p className="text-lg text-slate-400 -mt-8">Select a verb and tense to start your quiz.</p>
    <div>
      <label htmlFor="verb-select" className="block text-lg font-medium text-slate-300 mb-2">1. Choose a Verb</label>
      <select
        id="verb-select"
        value={selectedVerb}
        onChange={(e) => setSelectedVerb(e.target.value)}
        className="w-full p-3 bg-slate-700 border border-slate-600 rounded-lg shadow-sm focus:ring-2 focus:ring-indigo-500 focus:border-indigo-500 transition text-white"
      >
        {verbs.map(verb => <option key={verb.value} value={verb.value}>{verb.label}</option>)}
      </select>
    </div>
    <div>
      <label htmlFor="tense-select" className="block text-lg font-medium text-slate-300 mb-2">2. Choose a Tense to Test</label>
      <select
        id="tense-select"
        value={selectedTense}
        onChange={(e) => setSelectedTense(e.target.value)}
        className="w-full p-3 bg-slate-700 border border-slate-600 rounded-lg shadow-sm focus:ring-2 focus:ring-indigo-500 focus:border-indigo-500 transition text-white"
      >
        {tenses.map(tense => <option key={tense.value} value={tense.value}>{tense.label}</option>)}
      </select>
    </div>
    <button
      onClick={onStartQuiz}
      className="w-full bg-indigo-600 text-white font-bold py-4 px-6 rounded-lg hover:bg-indigo-500 focus:outline-none focus:ring-4 focus:ring-indigo-500 focus:ring-opacity-50 transition-transform transform hover:scale-105 shadow-lg"
    >
      Generate Quiz!
    </button>
  </div>
);

const QuizView = ({ questionData, currentIndex, totalQuestions, userAnswer, onAnswerSelect, onNext }) => {
    const [showHint, setShowHint] = useState(false);

    useEffect(() => {
        setShowHint(false); // Hide hint when question changes
    }, [questionData]);

    return (
        <div className="animate-fade-in">
            <div className="text-right text-slate-400 font-mono mb-4">{currentIndex + 1} / {totalQuestions}</div>
            <p className="font-semibold text-xl sm:text-2xl text-slate-100 mb-6 h-16">{questionData.question}</p>
            <div className="space-y-3">
                {questionData.options.map((option, i) => (
                    <button
                        type="button"
                        key={i}
                        onClick={() => onAnswerSelect(questionData.id, option)}
                        className={`w-full p-4 text-left rounded-lg border-2 transition-all duration-200 text-lg ${
                            userAnswer === option
                                ? 'bg-indigo-600 border-indigo-500 text-white shadow-md ring-2 ring-indigo-400'
                                : 'bg-slate-700 border-slate-600 hover:bg-slate-600 hover:border-slate-500'
                        }`}
                    >
                        <span className="font-mono mr-3 text-slate-400">{String.fromCharCode(65 + i)}.</span>{option}
                    </button>
                ))}
            </div>
            <div className="mt-6 flex justify-between items-center">
                 <div className="h-10">
                    <button onClick={() => setShowHint(!showHint)} className="text-indigo-400 hover:text-indigo-300 transition text-sm">
                        {showHint ? 'Hide Hint' : 'Show Hint'}
                    </button>
                    {showHint && (
                        <div className="text-sm text-amber-400 p-2 bg-slate-700/50 rounded-md mt-2 animate-fade-in-fast">
                            Target Tense: <span className="font-bold">{questionData.hint}</span>
                        </div>
                    )}
                </div>
                <button
                    onClick={onNext}
                    disabled={!userAnswer}
                    className="bg-indigo-600 text-white font-bold py-3 px-8 rounded-lg hover:bg-indigo-500 focus:outline-none focus:ring-4 focus:ring-indigo-500 focus:ring-opacity-50 transition-transform transform hover:scale-105 shadow-lg disabled:bg-slate-600 disabled:cursor-not-allowed disabled:transform-none"
                >
                    {currentIndex === totalQuestions - 1 ? 'Finish' : 'Next'}
                </button>
            </div>
        </div>
    );
};

const ResultsDisplay = ({ quizData, userAnswers, onStartOver }) => {
    const score = quizData.filter(q => userAnswers[q.id] === q.correctAnswer).length;
    const total = quizData.length;
    const percentage = total > 0 ? Math.round((score / total) * 100) : 0;

    return (
        <div className="space-y-6 animate-fade-in">
            <div className="text-center p-6 bg-slate-900/50 rounded-xl ring-1 ring-slate-700">
                <h2 className="text-3xl font-bold text-white">Quiz Complete!</h2>
                <p className="text-6xl font-bold mt-2" style={{color: percentage >= 80 ? '#4ade80' : percentage >= 50 ? '#facc15' : '#f87171'}}>
                    {percentage}%
                </p>
                <p className="text-xl text-slate-300 mt-1">You scored {score} out of {total}</p>
            </div>

            <div className="space-y-4">
                {quizData.map((q, index) => {
                    const userAnswer = userAnswers[q.id];
                    const isCorrect = userAnswer === q.correctAnswer;
                    const resultColor = isCorrect ? 'border-green-500 bg-green-900/20' : 'border-red-500 bg-red-900/20';
                    
                    return (
                        <div key={q.id} className={`p-4 border-l-4 rounded-r-lg ${resultColor}`}>
                            <p className="font-semibold text-slate-300">
                                <span className="font-bold mr-2">{index + 1}.</span>{q.question}
                            </p>
                            {isCorrect ? (
                                <p className="mt-2 text-green-400">
                                    <span className="font-bold">✓ Your answer: </span>{userAnswer}
                                </p>
                            ) : (
                                <>
                                    <p className="mt-2 text-red-400">
                                        <span className="font-bold">✗ Your answer: </span>{userAnswer || <span className="italic">No answer</span>}
                                    </p>
                                    <p className="mt-1 text-green-400">
                                        <span className="font-bold">Correct answer: </span>{q.correctAnswer}
                                    </p>
                                </>
                            )}
                        </div>
                    );
                })}
            </div>
            
            <button
                onClick={onStartOver}
                className="w-full bg-indigo-600 text-white font-bold py-4 px-6 rounded-lg hover:bg-indigo-500 focus:outline-none focus:ring-4 focus:ring-indigo-500 focus:ring-opacity-50 transition-transform transform hover:scale-105 shadow-lg"
            >
                Start a New Quiz
            </button>
        </div>
    );
};

    <div className="flex flex-col items-center justify-center p-12 text-center animate-fade-in">
        <svg className="animate-spin -ml-1 mr-3 h-12 w-12 text-indigo-400" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">
            <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4"></circle>
            <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>
        </svg>
        <p className="mt-4 text-lg font-medium text-slate-300">Crafting your custom verb quiz...</p>
        <p className="text-slate-400">This may take a moment.</p>
    </div>
);

const Footer = () => (
    <footer className="text-center mt-8">
        <p className="text-sm text-slate-500">A focused quiz tool to master French verb tenses.</p>
    </footer>
);
